    function buildFibonacciSequence(quantity) {
        if (!quantity) {
            return '';
        }
        if (quantity == 1) {
            return [0];
        }
        let arr = [0, 1];
        for (let i = 2; i < quantity; i++) {
            arr[i] = arr[i - 1] + arr[i - 2];
        };
        return arr;
    };
